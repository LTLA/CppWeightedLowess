204 /* Determining the `delta`. For a anchor point with x-coordinate `x`, we skip all
205 * points in `[x, x + delta]` before finding the next anchor point.
206 * We try to choose a `delta` that satisfies the constraints on the number
207 * of anchor points in `points`. A naive approach would be to simply divide the
208 * range of `x` by `points - 1`. However, this may place anchor points inside
209 * large gaps on the x-axis intervals where there are no actual observations.
210 *
211 * Instead, we try to distribute the anchor points so that they don't fall
212 * inside such large gaps. We do so by looking at the largest gaps and seeing
213 * what happens if we were to shift the anchor points to avoid such gaps. If we
214 * jump across a gap, though, we need to "use up" a anchor point to restart
215 * the sequence of anchor points on the other side of the gap. This requires some
216 * iteration to find the compromise that minimizes the 'delta' (and thus the
217 * degree of approximation in the final lowess calculation).
218 */
219 static Data_t derive_delta(int points, size_t n, const Data_t* x) {
220 std::vector<Data_t> diffs(n - 1);
221 for (size_t i = 0; i < diffs.size(); ++i) {
222 diffs[i] = x[i + 1] - x[i];
223 }
224
225 std::sort(diffs.begin(), diffs.end());
226 for (size_t i = 1; i < diffs.size(); ++i) {
227 diffs[i] += diffs[i-1];
228 }
229
230 Data_t lowest_delta = diffs.back();
231 for (size_t nskips = 0; nskips < points - 1 && nskips < diffs.size() - 1; ++nskips) {
232 Data_t candidate_delta = diffs[diffs.size() - nskips - 1] / (points - nskips);
233 lowest_delta = std::min(candidate_delta, lowest_delta);
234 }
235
236 return lowest_delta;
237 }

241 * a anchor point with x-coordinate `x`, we skip all points in `[x, x +

244 static void find_anchors(size_t n, const Data_t* x, Data_t d, std::vector<size_t>& anchors) {
245 anchors.clear();

257 return;
258 }

266 /* This function identifies the start and end index in the span for each chosen sampling
267 * point. It returns two arrays via reference containing said indices. It also returns
268 * an array containing the maximum distance between points at each span.
269 *
270 * We don't use the update-based algorithm in Cleveland's paper, as it ceases to be
271 * numerically stable once you throw in Data_t-precision weights. It's not particularly
272 * amenable to updating through cycles of addition and subtraction. At any rate, the
273 * algorithm as a whole remains quadratic (as weights must be recomputed) so there's no
274 * damage to scalability.
275 */
276 static std::vector<window> find_limits(const std::vector<size_t>& anchors,
277 Data_t spanweight,
278 size_t n,
279 const Data_t* x,
280 const Data_t* weights,
281 Data_t min_width)
282 {
283 const size_t nanchors = anchors.size();
284 std::vector<window> limits(nanchors);
285 auto half_min_width = min_width / 2;
286
287 for (size_t s = 0; s < nanchors; ++s) {
288 auto curpt = anchors[s], left = curpt, right = curpt;
289 auto curx = x[curpt];
290 Data_t curw = (weights == NULL ? 1 : weights[curpt]);
291 bool ende = (curpt == n - 1), ends = (curpt == 0);
292 Data_t mdist=0, ldist=0, rdist=0;
293
294 while (curw < spanweight && (!ende || !ends)) {
295 if (!ends) {
296 ldist = curx - x[left - 1];
297 }
298 if (!ende) {
299 rdist = x[right + 1] - curx;
300 }
301
302 // Go backwards if the left edge is not at the start already
303 // and the left distance is lower (or the right edge is at the end).
304 bool go_back = false;
305 if (!ends) {
306 if (ende || ldist <= rdist) {
307 go_back = true;
308 }
309 }
310
311 if (go_back) {
312 /* Move the span backwards. */
313 --left;
314
315 if (weights != NULL) {
316 curw += weights[left];
317 } else {
318 ++curw;
319 }
320
321 if (left==0) {
322 ends = true;
323 }
324
325 if (mdist < ldist) {
326 mdist = ldist;
327 }
328 } else {
329 /* Move the span forwards. */
330 ++right;
331
332 if (weights != NULL) {
333 curw += weights[right];
334 } else {
335 ++curw;
336 }
337
338 if (right == n - 1) {
339 ende = true;
340 }
341
342 if (mdist < rdist) {
343 mdist = rdist;
344 }
345 }
346 }
347
348 /* Once we've found the span, we stretch it out to include all ties. */
349 while (left > 0 && x[left] == x[left-1]) {
350 --left;
351 }
352
353 while (right < n - 1 && x[right]==x[right+1]) {
354 ++right;
355 }
356
357 /* Forcibly extending the span if it fails the min width. We use
358 * the existing 'left' and 'right' to truncate the search space.
359 */
360 if (mdist < half_min_width) {
361 left = std::lower_bound(x, x + left, curx - half_min_width) - x;
362
363 /* 'right' still refers to a point inside the window, and we
364 * already know that the window is too small, so we shift it
365 * forward by one to start searching outside. However,
366 * upper_bound gives us the first element that is _outside_ the
367 * window, so we need to subtract one to get to the last
368 * element _inside_ the window.
369 */
370 right = std::upper_bound(x + right + 1, x + n, curx + half_min_width) - x;
371 --right;
372
373 mdist = std::max(curx - x[left], x[right] - curx);
374 }
375
376 limits[s].left = left;
377 limits[s].right = right;
378 limits[s].distance = mdist;
379 }
380
381 return limits;
382 }

389 /* Computes the lowess fit at a given point using linear regression with a
390 * combination of tricube, prior and robustness weighting.
391 */
392 static Data_t lowess_fit (const size_t curpt, const window& limits,
393 size_t n,
394 const Data_t* x,
395 const Data_t* y,
396 const Data_t* weights,
397 const Data_t* robust_weights,
398 Data_t* work)
399 {
400 size_t left = limits.left, right = limits.right;
401 Data_t dist = limits.distance;
402
403 if (dist <= 0) {
404 Data_t ymean = 0, allweight = 0;
405 for (size_t pt = left; pt <= right; ++pt) {
406 Data_t curweight = robust_weights[pt];
407 if (weights != NULL) {
408 curweight *= weights[pt];
409 }
410 ymean += y[pt] * curweight;
411 allweight += curweight;
412 }
413
414 if (allweight == 0) { // ignore the robustness weights.
415 for (size_t pt = left; pt <= right; ++pt) {
416 Data_t curweight = 1;
417 if (weights != NULL) {
418 curweight = weights[pt];
419 }
420 ymean += y[pt] * curweight;
421 allweight += curweight;
422 }
423 }
424
425 ymean /= allweight;
426 return ymean;
427 }
428
429 Data_t xmean = 0, ymean = 0, allweight = 0;
430 for (size_t pt = left; pt <= right; ++pt) {
431 Data_t& current = work[pt];
432 current = cube(1 - cube(std::abs(x[curpt] - x[pt])/dist)) * robust_weights[pt];
433 if (weights != NULL) {
434 current *= weights[pt];
435 }
436 xmean += current * x[pt];
437 ymean += current * y[pt];
438 allweight += current;
439 }
440
441 if (allweight == 0) { // ignore the robustness weights.
442 for (size_t pt = left; pt <= right; ++pt) {
443 Data_t& current = work[pt];
444 current = cube(1 - cube(std::abs(x[curpt] - x[pt])/dist));
445 if (weights != NULL) {
446 current *= weights[pt];
447 }
448 xmean += current * x[pt];
449 ymean += current * y[pt];
450 allweight += current;
451 }
452 }
453
454 xmean /= allweight;
455 ymean /= allweight;
456
457 Data_t var=0, covar=0;
458 for (size_t pt = left; pt <= right; ++pt) {
459 Data_t temp = x[pt] - xmean;
460 var += temp * temp * work[pt];
461 covar += temp * (y[pt] - ymean) * work[pt];
462 }
463
464 // Still possible for var = 0 if all other points have zero weight.
465 if (var == 0) {
466 return ymean;
467 } else {
468 const Data_t slope = covar / var;
469 const Data_t intercept = ymean - slope * xmean;
470 return slope * x[curpt] + intercept;
471 }
472 }

479 /* This is a C version of the local weighted regression (lowess) trend fitting algorithm,
480 * based on the Fortran code in lowess.f from http://www.netlib.org/go written by Cleveland.
481 * Consideration of non-equal prior weights is added to the span calculations and linear
482 * regression. These weights are intended to have the equivalent effect of frequency weights
483 * (at least, in the integer case; extended by analogy to all non-negative values).
484 */
485 void robust_lowess(size_t n,
486 const Data_t* x,
487 const Data_t* y,
488 const Data_t* weights,
489 Data_t* fitted,
490 Data_t* robust_weights)
491 const {
492 std::vector<Data_t> workspace(n);
493
494 /* Computing the span weight that each span must achieve. */
495 const Data_t totalweight = (weights != NULL ? std::accumulate(weights, weights + n, 0.0) : n);
496 const Data_t spanweight = (span_as_proportion ? span * totalweight : span);
497
498 /* Finding the anchors. If we're using the weights as frequencies, we
499 * pass them in when we're looking for the span limits for each anchor.
500 */
501 std::vector<size_t> anchors;
502 if (points < n || delta > 0) {
503 Data_t eff_delta = (delta < 0 ? derive_delta(points, n, x) : delta);
504 find_anchors(n, x, eff_delta, anchors);
505 } else {
506 anchors.resize(n);
507 std::iota(anchors.begin(), anchors.end(), 0);
508 }
509
510 auto limits = find_limits(anchors, spanweight, n, x, (freqweights ? weights : NULL), min_width);
511
512 /* Setting up the robustness weights, if robustification is requested.
513 */
514 std::fill(robust_weights, robust_weights + n, 1);
515 Data_t min_mad = 0;
516 std::vector<size_t> residual_permutation;
517 if (iterations) {
518 residual_permutation.resize(n);
519
520 /* We use the range so guarantee that we catch the scale. Otherwise
521 * if we used the MAD of 'y', it could be conceivable that we would
522 * end up with a min_mad of zero again, e.g., if the majority of
523 * points have the same value. In contrast, if the range is zero,
524 * we just quit early.
525 */
526 Data_t range = (*std::max_element(y, y + n) - *std::min_element(y, y + n));
527 if (range == 0) {
528 std::copy(y, y + n, fitted);
529 return;
530 }
531 min_mad = 0.00000000001 * range;
532 }
533
534 for (int it = 0; it <= iterations; ++it) { // Robustness iterations.
535 fitted[0] = lowess_fit(0, limits[0], n, x, y, weights, robust_weights, workspace.data());
536 size_t last_anchor = 0;
537
538 for (size_t s = 1; s < anchors.size(); ++s) { // fitted values for anchor points, interpolating the rest.
539 auto curpt = anchors[s];
540 fitted[curpt] = lowess_fit(curpt, limits[s], n, x, y, weights, robust_weights, workspace.data());
541
542 if (curpt - last_anchor > 1) {
543 /* Some protection is provided against infinite slopes. This shouldn't be
544 * a problem for non-zero delta; the only concern is at the final point
545 * where the covariate distance may be zero.
546 */
547 Data_t current = x[curpt] - x[last_anchor];
548 if (current > 0) {
549 const Data_t slope = (fitted[curpt] - fitted[last_anchor])/current;
550 const Data_t intercept = fitted[curpt] - slope * x[curpt];
551 for (size_t subpt = last_anchor + 1; subpt < curpt; ++subpt) {
552 fitted[subpt] = slope * x[subpt] + intercept;
553 }
554 } else {
555 const Data_t ave = (fitted[curpt] + fitted[last_anchor]) / 2;
556 for (size_t subpt = last_anchor + 1; subpt < curpt; ++subpt) {
557 fitted[subpt] = ave;
558 }
559 }
560 }
561
562 last_anchor = curpt;
563 }
564
565 if (it < iterations) {
566 /* Computing the weighted MAD of the absolute values of the residuals. */
567 for (size_t pt = 0; pt < n; ++pt) {
568 workspace[pt] = std::abs(y[pt] - fitted[pt]);
569 }
570
571 std::iota(residual_permutation.begin(), residual_permutation.end(), 0);
572 std::sort(residual_permutation.begin(), residual_permutation.end(),
573 [&](size_t left, size_t right) -> bool {
574 return workspace[left] < workspace[right];
575 }
576 );
577
578 Data_t curweight = 0;
579 Data_t cmad = 100;
580 const Data_t halfweight = totalweight/2;
581
582 for (size_t i = 0; i < n; ++i) {
583 auto pt = residual_permutation[i];
584 if (weights != NULL) {
585 curweight += weights[pt];
586 } else {
587 ++curweight;
588 }
589
590 if (curweight == halfweight) { // exact match, need to take the median.
591 cmad = 3 * (workspace[pt] + workspace[residual_permutation[i+1]]);
592 break;
593 } else if (curweight > halfweight) {
594 cmad = 6 * workspace[pt];
595 break;
596 }
597 }
598
599 /* Both limma::weightedLowess and the original Fortran code have an early
600 * termination condition that stops the robustness iterations when the MAD
601 * is small. We do not implement this and just allow the specified number of
602 * iterations to run, as the termination can fail in pathological examples
603 * where a minority of points are affected by an outlier. In such cases,
604 * the MAD may indeed be very small as most residuals are fine, and we would
605 * fail to robustify against the few outliers.
606 */
607
608 cmad = std::max(cmad, min_mad); // avoid difficulties from numerical precision when all residuals are theoretically zero.
609 for (size_t i =0; i < n; ++i) {
610 if (workspace[i] < cmad) {
611 robust_weights[i] = square(1 - square(workspace[i]/cmad));
612 } else {
613 robust_weights[i] = 0;
614 }
615 }
616 }
617 }
618
619 return;
620 }

623 void sort_and_run(size_t n, Data_t* x, Data_t* y, Data_t* weights, Data_t* fitted, Data_t* robust_weights) const {
624 std::vector<size_t> permutation(n);
625 std::iota(permutation.begin(), permutation.end(), 0);
626 std::sort(permutation.begin(), permutation.end(), [&](size_t left, size_t right) -> bool {
627 return x[left] < x[right];
628 });
629
630 // Reordering values in place.
631 std::vector<uint8_t> used(n);
632 std::fill(used.begin(), used.end(), 0);
633 for (size_t i = 0; i < permutation.size(); ++i) {
634 if (used[i]) {
635 continue;
636 }
637 used[i] = 1;
638
639 size_t current = i, replacement = permutation[i];
640 while (replacement != i) {
641 std::swap(x[current], x[replacement]);
642 std::swap(y[current], y[replacement]);
643 if (weights != NULL) {
644 std::swap(weights[current], weights[replacement]);
645 }
646
647 current = replacement;
648 used[replacement] = 1;
649 replacement = permutation[replacement];
650 }
651 }
652
653 // Computing the fitted values and robustness weights.
654 Data_t* rptr = robust_weights;
655 std::vector<Data_t> rbuffer;
656 if (robust_weights == NULL) {
657 rbuffer.resize(n);
658 rptr = rbuffer.data();
659 }
660 robust_lowess(n, x, y, weights, fitted, rptr);
661
662 // Unpermuting the fitted values in place.
663 std::fill(used.begin(), used.end(), 0);
664 for (size_t i = 0; i < permutation.size(); ++i) {
665 if (used[i]) {
666 continue;
667 }
668 used[i] = 1;
669
670 size_t replacement = permutation[i];
671 while (replacement != i) {
672 std::swap(fitted[i], fitted[replacement]);
673 if (robust_weights) {
674 std::swap(robust_weights[i], robust_weights[replacement]);
675 }
676
677 used[replacement] = 1;
678 replacement = permutation[replacement];
679 }
680 }
681
682 return;
683 }

697 void run(size_t n, const Data_t* x, const Data_t* y, const Data_t* weights, Data_t* fitted, Data_t* robust_weights) const {
698 if (sorted) {
699 std::vector<Data_t> rbuffer;
700 if (robust_weights == NULL) {
701 rbuffer.resize(n);
702 robust_weights = rbuffer.data();
703 }
704 robust_lowess(n, x, y, weights, fitted, robust_weights);
705 } else {
706 /* Sorts the observations by the means, applies the same permutation to the
707 * variances. This makes downstream processing quite a lot easier.
708 */
709 std::vector<Data_t> xbuffer(x, x+ n);
710 std::vector<Data_t> ybuffer(y, y + n);
711
712 std::vector<Data_t> wbuffer;
713 Data_t* wptr = NULL;
714 if (weights) {
715 wbuffer.insert(wbuffer.end(), weights, weights + n);
716 wptr = wbuffer.data();
717 }
718
719 sort_and_run(n, xbuffer.data(), ybuffer.data(), wptr, fitted, robust_weights);
720 }
721 return;
