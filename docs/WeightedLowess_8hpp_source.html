127  /* Determining the `delta`. For a anchor point with x-coordinate `x`, we skip all
128  * points in `[x, x + delta]` before finding the next anchor point.
129  * We try to choose a `delta` that satisfies the constraints on the number
130  * of anchor points in `points`. A naive approach would be to simply divide the
131  * range of `x` by `points - 1`. However, this may place anchor points inside
132  * large gaps on the x-axis intervals where there are no actual observations.
133  *
134  * Instead, we try to distribute the anchor points so that they don't fall
135  * inside such large gaps. We do so by looking at the largest gaps and seeing
136  * what happens if we were to shift the anchor points to avoid such gaps. If we
137  * jump across a gap, though, we need to "use up" a anchor point to restart
138  * the sequence of anchor points on the other side of the gap. This requires some
139  * iteration to find the compromise that minimizes the 'delta' (and thus the
140  * degree of approximation in the final lowess calculation).
141  */
142  static double derive_delta(int points, size_t n, const double* x, std::vector<double>& diffs) {
143  diffs.resize(n - 1);
144  for (size_t i = 0; i < diffs.size(); ++i) {
145  diffs[i] = x[i + 1] - x[i];
146  }
147 
148  std::sort(diffs.begin(), diffs.end());
149  for (size_t i = 1; i < diffs.size(); ++i) {
150  diffs[i] += diffs[i-1];
151  }
152 
153  double lowest_delta = diffs.back();
154  for (size_t nskips = 0; nskips < points - 1 && nskips < diffs.size() - 1; ++nskips) {
155  double candidate_delta = diffs[diffs.size() - nskips - 1] / (points - nskips);
156  lowest_delta = std::min(candidate_delta, lowest_delta);
157  }
158 
159  return lowest_delta;
160  }

193  /* This function identifies the start and end index in the span for each chosen sampling
194  * point. It returns two arrays via reference containing said indices. It also returns
195  * an array containing the maximum distance between points at each span.
196  *
197  * We don't use the update-based algorithm in Cleveland's paper, as it ceases to be
198  * numerically stable once you throw in double-precision weights. It's not particularly
199  * amenable to updating through cycles of addition and subtraction. At any rate, the
200  * algorithm as a whole remains quadratic (as weights must be recomputed) so there's no
201  * damage to scalability.
202  */
203  static void find_limits(const std::vector<size_t>& anchors,
204  double spanweight,
205  size_t n,
206  const double* x,
207  const double* weights,

211  limits.resize(nanchors);
212 
213  for (size_t s = 0; s < nanchors; ++s) {
214  auto curpt = anchors[s], left = curpt, right = curpt;
215  double curw = (weights == NULL ? 1 : weights[curpt]);
216  bool ende = (curpt == n - 1), ends = (curpt == 0);
217  double mdist=0, ldist=0, rdist=0;
218 
219  while (curw < spanweight && (!ende || !ends)) {
220  if (!ends) {

222  }
223  if (!ende) {

225  }
226

228  if (!ends) {
229  if (ende || ldist <= rdist) {
230  --left;
231 
232  if (weights != NULL) {
233  curw += weights[left];
234  } else {
235  ++curw;
236  }
237 
238  if (left==0) {
239  ends=1;
240  }
241 
242  if (mdist < ldist) {
243  mdist=ldist;
244  }
245  }
246  }
247 
248  /* Move the span forwards. */
249  if (!ende) {
250  if (ends || ldist > rdist) {
251  ++right;
252

254  curw += weights[right];
255  } else {

257  }
258

261  }
262

264  mdist=rdist;
265  }
266  }
267  }
268  }
269

271  while (left > 0 && x[left] == x[left-1]) {
272  --left;
273  }
274 
275  while (right < n - 1 && x[right]==x[right+1]) {
276  ++right;
277  }
278 
279  limits[s].left = left;
280  limits[s].right = right;
281  limits[s].distance = mdist;
282  }

291  /* Computes the lowess fit at a given point using linear regression with a
292  * combination of tricube, prior and robustness weighting.
293  */
294  static double lowess_fit (const size_t curpt, const window& limits,
295  size_t n,
296  const double* x,
297  const double* y,
298  const double* weights,
299  const double* robust_weights,
300  double* work)
301  {
302  size_t left = limits.left, right = limits.right;
303  double dist = limits.distance;
304 
305  if (dist <= 0) {
306  double ymean = 0, allweight = 0;
307  for (size_t pt = left; pt <= right; ++pt) {
308  double curweight = robust_weights[pt];
309  if (weights != NULL) {
310  curweight *= weights[pt];
311  }
312  ymean += y[pt] * curweight;
313  allweight += curweight;
314  }
315 
316  if (allweight == 0) { // ignore the robustness weights.
317  for (size_t pt = left; pt <= right; ++pt) {
318  double curweight = 1;
319  if (weights != NULL) {
320  curweight = weights[pt];
321  }
322  ymean += y[pt] * curweight;
323  allweight += curweight;
324  }
325  }
326 
327  ymean /= allweight;
328  return ymean;
329  }
330 
331  double xmean = 0, ymean = 0, allweight = 0;
332  for (size_t pt = left; pt <= right; ++pt) {
333  double& current = work[pt];
334  current = cube(1 - cube(std::abs(x[curpt] - x[pt])/dist)) * robust_weights[pt];
335  if (weights != NULL) {
336  current *= weights[pt];
337  }
338  xmean += current * x[pt];
339  ymean += current * y[pt];
340  allweight += current;
341  }
342 
343  if (allweight == 0) { // ignore the robustness weights.
344  for (size_t pt = left; pt <= right; ++pt) {
345  double& current = work[pt];
346  current = cube(1 - cube(std::abs(x[curpt] - x[pt])/dist));
347  if (weights != NULL) {
348  current *= weights[pt];
349  }
350  xmean += current * x[pt];
351  ymean += current * y[pt];
352  allweight += current;
353  }
354  }
355 
356  xmean /= allweight;
357  ymean /= allweight;
358 
359  double var=0, covar=0;
360  for (size_t pt = left; pt <= right; ++pt) {
361  double temp = x[pt] - xmean;
362  var += temp * temp * work[pt];
363  covar += temp * (y[pt] - ymean) * work[pt];
364  }
365 
366  // Still possible for var = 0 if all other points have zero weight.
367  if (var == 0) {
368  return ymean;
369  } else {
370  const double slope = covar / var;
371  const double intercept = ymean - slope * xmean;
372  return slope * x[curpt] + intercept;
373  }

384  /* This is a C version of the local weighted regression (lowess) trend fitting algorithm,
385  * based on the Fortran code in lowess.f from http://www.netlib.org/go written by Cleveland.
386  * Consideration of non-equal prior weights is added to the span calculations and linear
387  * regression. These weights are intended to have the equivalent effect of frequency weights
388  * (at least, in the integer case; extended by analogy to all non-negative values).
389  */
390  void robust_lowess(size_t n,
391  const double* x,
392  const double* y,
393  const double* weights,
394  double* fitted,
395  double* robust_weights)
396  {
397  workspace.resize(n);
398 
399  /* Computing the span weight that each span must achieve. */
400  double totalweight = 0;
401  if (weights != NULL) {
402  totalweight = std::accumulate(weights, weights + n, 0.0);
403  } else {
404  totalweight = n;
405  }
406  const double spanweight = totalweight * span;
407 
408  /* Finding the anchors. If we're using the weights as frequencies, we
409  * pass them in when we're looking for the span limits for each anchor.
410  */
411  if (points < n || delta > 0) {
412  double eff_delta = (delta < 0 ? derive_delta(points, n, x, diffs) : delta);
413  find_anchors(n, x, eff_delta, anchors);
414  } else {
415  anchors.resize(n);
416  std::iota(anchors.begin(), anchors.end(), 0);
417  }
418  find_limits(anchors, spanweight, n, x, (freqweights ? weights : NULL), limits);
419 
420  /* Setting up the robustness weights, if robustification is requested.
421  */
422  std::fill(robust_weights, robust_weights + n, 1);
423  double min_mad = 0;
424  if (iterations) {
425  residual_permutation.resize(n);
426 
427  /* We use the range so guarantee that we catch the scale. Otherwise
428  * if we used the MAD of 'y', it could be conceivable that we would
429  * end up with a min_mad of zero again, e.g., if the majority of
430  * points have the same value. In contrast, if the range is zero,
431  * we just quit early.
432  */
433  double range = (*std::max_element(y, y + n) - *std::min_element(y, y + n));
434  if (range == 0) {
435  std::copy(y, y + n, fitted);
436  return;
437  }
438  min_mad = 0.00000000001 * range;
439  }
440 
441  for (int it = 0; it <= iterations; ++it) { // Robustness iterations.
442  fitted[0] = lowess_fit(0, limits[0], n, x, y, weights, robust_weights, workspace.data());
443  size_t last_anchor = 0;
444 
445  for (size_t s = 1; s < anchors.size(); ++s) { // fitted values for anchor points, interpolating the rest.
446  auto curpt = anchors[s];
447  fitted[curpt] = lowess_fit(curpt, limits[s], n, x, y, weights, robust_weights, workspace.data());
448 
449  if (curpt - last_anchor > 1) {
450  /* Some protection is provided against infinite slopes. This shouldn't be
451  * a problem for non-zero delta; the only concern is at the final point
452  * where the covariate distance may be zero.
453  */
454  double current = x[curpt] - x[last_anchor];
455  if (current > 0) {
456  const double slope = (fitted[curpt] - fitted[last_anchor])/current;
457  const double intercept = fitted[curpt] - slope * x[curpt];
458  for (size_t subpt = last_anchor + 1; subpt < curpt; ++subpt) {
459  fitted[subpt] = slope * x[subpt] + intercept;
460  }
461  } else {
462  const double ave = (fitted[curpt] + fitted[last_anchor]) / 2;
463  for (size_t subpt = last_anchor + 1; subpt < curpt; ++subpt) {
464  fitted[subpt] = ave;
465  }
466  }
467  }
468 
469  last_anchor = curpt;
470  }
471 
472  if (it < iterations) {
473  /* Computing the weighted MAD of the absolute values of the residuals. */
474  for (size_t pt = 0; pt < n; ++pt) {
475  workspace[pt] = std::abs(y[pt] - fitted[pt]);
476  }
477 
478  std::iota(residual_permutation.begin(), residual_permutation.end(), 0);
479  std::sort(residual_permutation.begin(), residual_permutation.end(),
480  [&](size_t left, size_t right) -> bool {
481  return workspace[left] < workspace[right];
482  }
483  );
484 
485  double curweight = 0;
486  double cmad = 100;
487  const double halfweight = totalweight/2;
488 
489  for (size_t i = 0; i < n; ++i) {
490  auto pt = residual_permutation[i];
491  if (weights != NULL) {
492  curweight += weights[pt];
493  } else {
494  ++curweight;
495  }
496 
497  if (curweight == halfweight) { // exact match, need to take the median.
498  cmad = 3 * (workspace[pt] + workspace[residual_permutation[i+1]]);
499  break;
500  } else if (curweight > halfweight) {
501  cmad = 6 * workspace[pt];
502  break;
503  }
504  }
505 
506  /* Both limma::weightedLowess and the original Fortran code have an early
507  * termination condition that stops the robustness iterations when the MAD
508  * is small. We do not implement this and just allow the specified number of
509  * iterations to run, as the termination can fail in pathological examples
510  * where a minority of points are affected by an outlier. In such cases,
511  * the MAD may indeed be very small as most residuals are fine, and we would
512  * fail to robustify against the few outliers.
513  */
514 
515  cmad = std::max(cmad, min_mad); // avoid difficulties from numerical precision when all residuals are theoretically zero.
516  for (size_t i =0; i < n; ++i) {
517  if (workspace[i] < cmad) {
518  robust_weights[i] = square(1 - square(workspace[i]/cmad));
519  } else {
520  robust_weights[i] = 0;
521  }
522  }
523  }
524  }
525 
526  return;
527  }

535  permutation.resize(n);
536  std::iota(permutation.begin(), permutation.end(), 0);
537  std::sort(permutation.begin(), permutation.end(), [&](size_t left, size_t right) -> bool {
538  return x[left] < x[right];
539  });

541  // Reordering values in place.
542  used.resize(n);
543  std::fill(used.begin(), used.end(), 0);
544  for (size_t i = 0; i < permutation.size(); ++i) {
545  if (used[i]) {
546  continue;
547  }
548  used[i] = 1;
549 
550  size_t current = i, replacement = permutation[i];
551  while (replacement != i) {
552  std::swap(x[current], x[replacement]);
553  std::swap(y[current], y[replacement]);
554  if (weights != NULL) {
555  std::swap(weights[current], weights[replacement]);
556  }
557 
558  current = replacement;
559  used[replacement] = 1;
560  replacement = permutation[replacement];
561  }
562  }
563 
564  // Computing the fitted values and robustness weights.
565  double* rptr = robust_weights;
566  if (robust_weights == NULL) {
567  rbuffer.resize(n);
568  rptr = rbuffer.data();
569  }
570  robust_lowess(n, x, y, weights, fitted, rptr);
571 
572  // Unpermuting the fitted values in place.
573  std::fill(used.begin(), used.end(), 0);
574  for (size_t i = 0; i < permutation.size(); ++i) {
575  if (used[i]) {
576  continue;
577  }
