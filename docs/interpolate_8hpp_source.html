60 AssignedSegments output;

64 {
65 const auto right = x_fit[anchors[0]];
66 while (counter < num_points_out && x_out[counter] < right) {
67 ++counter;
68 }
69 }

72 output.boundaries[i - 1] = counter;
73 if (counter == num_points_out) {

75 }
76 const auto right = x_fit[anchors[i]];
77 while (counter < num_points_out && x_out[counter] < right) {
78 ++counter;
79 }
80 }
81
82 // Anyone equal to the last anchor get assigned to the last segment.
83 {
84 const auto right = x_fit[anchors[num_anchors - 1]];
85 while (counter < num_points_out && x_out[counter] == right) {

87 }
88 output.boundaries[num_anchors - 1] = counter;
89 }

125template<typename Data_>
126void interpolate(
127 const Data_* const x_fit,
128 const PrecomputedWindows<Data_>& windows_fit,
129 const Data_* const fitted_fit,
130 const Data_* const x_out,
131 const AssignedSegments& assigned_out,
132 Data_* const fitted_out,
133 int num_threads
134) {
135 const auto& anchors = windows_fit.anchors;
136 const auto num_anchors = anchors.size();
137 const auto num_anchors_m1 = num_anchors - 1;
138
139 // One would think that we should parallelize across x_out instead of anchors,
140 // as this has better worksharing when x_out is not evenly distributed across anchor segments.
141 // However, if we did so, we'd have to store the slope and intercept for the anchor segments first,
142 // then look up the slope and intercept for each element of x_out.
143 // This involves an extra memory access and is not SIMD-able.
144 parallelize(num_threads, num_anchors_m1, [&](const int, const decltype(internal::I(num_anchors_m1)) start, const decltype(internal::I(num_anchors_m1)) length) {
145 for (decltype(internal::I(start)) s = start, end = start + length; s < end; ++s) {
146 const auto run_start = assigned_out.boundaries[s];
147 const auto run_end = assigned_out.boundaries[s + 1];
148 if (run_start == run_end) {
149 continue;
150 }
151
152 const auto left_anchor = windows_fit.anchors[s];
153 const auto right_anchor = windows_fit.anchors[s + 1];
154 const Data_ xdiff = x_fit[right_anchor] - x_fit[left_anchor];
155 const Data_ ydiff = fitted_fit[right_anchor] - fitted_fit[left_anchor];
156 if (xdiff > 0) {
157 const Data_ slope = ydiff / xdiff;
158 const Data_ intercept = fitted_fit[right_anchor] - slope * x_fit[right_anchor];
159 for (auto outpt = run_start; outpt < run_end; ++outpt) {
160 fitted_out[outpt] = slope * x_out[outpt] + intercept;
161 }
162 } else {
163 // Protect against infinite slopes by just taking the average.
164 const Data_ ave = fitted_fit[left_anchor] + ydiff / 2;
165 std::fill(fitted_out + run_start, fitted_out + run_end, ave);
166 }
167 }
168 });
169}
